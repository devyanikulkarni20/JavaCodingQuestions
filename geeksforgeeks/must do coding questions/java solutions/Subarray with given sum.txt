//Subarray with given sum

class Solution
{
    //Function to find a continuous sub-array which adds up to a given number.
    static ArrayList<Integer> subarraySum(int[] arr, int n, int sum) 
    {
        ArrayList<Integer> output = new ArrayList<>(2);
        int current_sum=arr[0],start=0,i;

        for(i = 1 ; i < n ; i++)
        {
            while(current_sum > sum && start < i-1){
                current_sum = current_sum - arr[start];
                start++;
            }

            if(current_sum == sum){
                output.add(start+1);
                output.add(i);
                return output;
            }
            // if (i < n)
            current_sum = current_sum + arr[i];
        }
        output.add(-1);
        return output;
    }
}
